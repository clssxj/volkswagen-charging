# 环境变量配置示例
# 复制此文件为 .env 并填入实际值

# ============================================
# 高德地图配置
# ============================================
# 高德地图Web端API Key
# 获取方式：https://lbs.amap.com/
# 1. 注册并登录高德开放平台
# 2. 进入"应用管理" -> "我的应用"
# 3. 创建新应用，选择"Web端(JS API)"
# 4. 获取API Key
VITE_AMAP_KEY=your_amap_key_here

# ============================================
# API配置
# ============================================
# API基础URL
# 开发环境建议使用本地地址
VITE_API_BASE_URL=http://localhost:8080/api

# 生产环境示例
# VITE_API_BASE_URL=https://api.your-domain.com/api

# ============================================
# WebSocket配置
# ============================================
# WebSocket服务器地址
# 开发环境
VITE_WS_URL=ws://localhost:8080/ws

# 生产环境示例
# VITE_WS_URL=wss://api.your-domain.com/ws

# ============================================
# Mock数据配置
# ============================================
# 是否使用Mock数据
# true: 使用Mock数据，无需后端服务
# false: 连接真实后端API
VITE_USE_MOCK=true

# ============================================
# 可选配置
# ============================================
# API请求超时时间（毫秒）
# VITE_API_TIMEOUT=10000

# 是否启用调试模式
# VITE_DEBUG=false

# Sentry错误监控DSN（可选）
# VITE_SENTRY_DSN=

# Google Analytics ID（可选）
# VITE_GA_ID=

# ============================================
# 注意事项
# ============================================
# 1. 所有以 VITE_ 开头的环境变量会被Vite注入到客户端代码中
# 2. 不要在环境变量中存储敏感信息（如私钥、密码等）
# 3. .env 文件不应提交到Git仓库（已在.gitignore中）
# 4. 不同环境可以创建不同的env文件：
#    - .env.development (开发环境)
#    - .env.production (生产环境)
#    - .env.local (本地覆盖，优先级最高)
